void TraiterChoix(int choix){
    int voyageurs, n;
    char nom[50], quartier[50], IdRecherche[20], id[20];
    station *temp;
  switch(choix) {
            //Section Ajout
            case 1:
                printf("ID: "); LireChaine(id, 20);
                printf("Nom: "); LireChaine(nom, 50);
                printf("Quartier: "); LireChaine(quartier, 50);
                printf("Voyageurs: "); voyageurs = LireInt();
                InsertionDebut(id, nom, quartier, voyageurs);
                pause();
                break;

            case 2:
                printf("ID: "); LireChaine(id, 20);
                printf("Nom: "); LireChaine(nom, 50);
                printf("Quartier: "); LireChaine(quartier, 50);
                printf("Voyageur: "); voyageurs = LireInt();
                InsertionFin(id, nom, quartier, voyageurs);
                pause();
                break;

            case 3:
                printf("Inserer apres ID: "); LireChaine(IdRecherche, 20);
                temp = RechercheId(IdRecherche);
                if(temp != NULL){
                printf("ID: "); LireChaine(id, 20);
                printf("Nom: "); LireChaine(nom, 50);
                printf("Quartier: "); LireChaine(quartier, 50);
                printf("Voyageurs: "); voyageurs = LireInt();
                InsertionApresId(id, nom, quartier, voyageurs, IdRecherche);

                }else{
                  printf("\n---ERREUR!--\n");
                 }
                pause();
                break;

            //Section Supprimer
            case 4: SupprimerHead(); pause(); break;
            case 5: SupprimerQueue(); pause(); break;
            case 6:
                printf("ID a supprimer: "); LireChaine(IdRecherche, 20);
                SupprimerId(IdRecherche);
                pause();
                break;

            //Section Recherche
            case 7:
                printf("ID a rechercher: "); LireChaine(IdRecherche, 20);
                temp = RechercheId(IdRecherche);
                AffichageSeul(temp);
                pause();
                break;

            case 8: AfficherQuartier(); pause(); break;

            //Section Affichage
            case 9: AffichageDebut(); pause(); break;
            case 10: AffichageFin(); pause(); break;

            //Section Statistique
            case 11: Statistique(); pause(); break;

            //Section save/load
            case 12: Sauvegarder(); pause(); break;
            case 13: LireFichier(); pause(); break;

            //Simulation
            case 14: printf("--Combien de station : "); n = LireInt();
                     AutoGenerateLOL(n); pause(); break;
             //Export
            case 15 : ExportRapport(); pause(); break;

            //Modification
            case 16: UpdateStation(); pause(); break;
            case 17: TrierCroissant(); pause(); break;


            case 0:
                printf("\n--AU REVOIR!--\n");
                break;

            default:
                printf("\n--CHOIX INVALIDE!--\n");
                pause();
        }

}
