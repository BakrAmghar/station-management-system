station *head = NULL;

//Fonction pr Creer une station:
station *CreerStation(char id[], char nom[], char quartier[], int voyageurs){
  station *nv = (station *)malloc(sizeof(station));

  nv->voyageurs = voyageurs;
  strcpy(nv->id, id);
  strcpy(nv->nom, nom);
  strcpy(nv->quartier, quartier);

  nv->next = NULL;
  nv->prev = NULL;

  return nv;
}

void Statistique(){
 int NbrVoyageurs = 0;
 int NbrStation = 0;
 station *temp = head;
 station *maxVoyageur = head;
 station *minVoyageur = head;

 if(temp == NULL) return;

   while(temp != NULL){
     NbrVoyageurs += temp->voyageurs;
     NbrStation++;

      if(temp->voyageurs > maxVoyageur->voyageurs) maxVoyageur = temp;
      if(temp->voyageurs < minVoyageur->voyageurs) minVoyageur = temp;

     temp = temp->next;
   }
   float Moy = NbrVoyageurs/NbrStation;
   printf("\n+========STATISTIQUE================================+\n");
   printf("| Nombre de station                : %-14d |\n",NbrStation);
   printf("| Total de Voyageurs               : %-14d |\n",NbrVoyageurs);
   printf("| Moyenne de voyageurs par station : %-14.2f |\n",Moy);
   printf("| Station la plus active           : %-14s |\n",maxVoyageur->id);
   printf("| Station la moins active          : %-14s |",minVoyageur->id);
   printf("\n+===================================================+\n");


  LogAction("STATS - System analysis completed");
}

void AutoGenerateLOL(int n){
char *quartiers[] = {
        "Tanger", "BniMakada", "Iberia", "Boulevard", "Sahat Madina", "Sahat Omam", "Dradeb"
    };

    char *noms[] = {
        "Station A1", "Station B1", "Station C1",
        "Station D1", "Station E1"
    };

    int qSize = 7;
    int nSize = 5;

    for(int i = 1; i <= n; i++){

        char id[20];
        sprintf(id, "ST%03d", i);

        char nom[50];
        strcpy(nom, noms[rand() % nSize]);

        char quartier[50];
        strcpy(quartier, quartiers[rand() % qSize]);

        int voyageurs = (rand() % 491) + 10;

        InsertionFin(id, nom, quartier, voyageurs);
    }

    printf("\n-- %d STATIONS GENEREES AVEC SUCCES --\n", n);

    LogAction("LOAD - Database imported successfully");
}

void TrierCroissant(){
  if(head == NULL || head->next == NULL){
        printf("\n-- RIEN A TRIER --\n");
        return;
    }

    station *i, *j;

    for(i = head; i != NULL; i = i->next){
        for(j = i->next; j != NULL; j = j->next){

            if(i->voyageurs > j->voyageurs){

                //data swap hhh
                char id[20], nom[50], quartier[50];
                int voy;

                // save
                strcpy(id, i->id);
                strcpy(nom, i->nom);
                strcpy(quartier, i->quartier);
                voy = i->voyageurs;

                //change
                strcpy(i->id, j->id);
                strcpy(i->nom, j->nom);
                strcpy(i->quartier, j->quartier);
                i->voyageurs = j->voyageurs;


                strcpy(j->id, id);
                strcpy(j->nom, nom);
                strcpy(j->quartier, quartier);
                j->voyageurs = voy;
            }
        }
    }

    printf("\n-- TRI PAR VOYAGEURS CROISSANT EFFECTUE --\n");
}
